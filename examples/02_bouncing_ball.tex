% Example 02: Bouncing Ball
% Parabolic trajectory with shadow on the ground plane.
% Two full bounces with a fading trail of past positions.
% Parameter: \PARAM sweeps 0 to 1 (default range), two full bounce cycles
% Render: tikzgif render 02_bouncing_ball.tex
% Difficulty: beginner
% Features demonstrated: mathematical expressions, shadows, trail effect

\documentclass[tikz]{standalone}
\usepackage{tikz}
\usepackage{xcolor}

% USC Brand Colors
\definecolor{garnet}{HTML}{73000A}
\definecolor{rose}{HTML}{CC2E40}
\definecolor{atlantic}{HTML}{466A9F}
\definecolor{congaree}{HTML}{1F414D}
\definecolor{horseshoe}{HTML}{65780B}
\definecolor{grass}{HTML}{CED318}
\definecolor{honeycomb}{HTML}{A49137}
\definecolor{warmgrey}{HTML}{676156}
\definecolor{sandstorm}{HTML}{FFF2E3}

\begin{document}
\begin{tikzpicture}
  \pgfmathsetmacro{\tval}{\PARAM}
  \useasboundingbox (-1,-1.2) rectangle (11,6);

  % Ground
  \fill[rose!20] (-0.5,-0.8) rectangle (10.5,0);
  \draw[thick] (-0.5,0) -- (10.5,0);

  % Trail: draw past positions as fading circles
  \foreach \k in {1,2,...,15} {
    \pgfmathsetmacro{\tpast}{max(\tval - \k * 0.012, 0)}
    \pgfmathsetmacro{\xpast}{\tpast * 10}
    \pgfmathsetmacro{\tlocal}{2*\tpast - floor(2*\tpast)}
    \pgfmathsetmacro{\trem}{max(0, min(1, (\tlocal - 0.02) / 0.96))}
    \pgfmathsetmacro{\ypast}{4 * 4 * \trem * (1 - \trem) + 0.35}
    \pgfmathsetmacro{\opac}{0.35 * (1 - \k/16)}
    \pgfmathsetmacro{\sz}{0.28 * (1 - \k/20)}
    \fill[garnet, opacity=\opac] (\xpast, \ypast) circle (\sz);
  }

  % Current position
  \pgfmathsetmacro{\xpos}{\tval * 10}
  \pgfmathsetmacro{\tlocal}{2*\tval - floor(2*\tval)}
  \pgfmathsetmacro{\trem}{max(0, min(1, (\tlocal - 0.02) / 0.96))}
  \pgfmathsetmacro{\ypos}{4 * 4 * \trem * (1 - \trem) + 0.35}

  % Ball
  \fill[garnet] (\xpos, \ypos) circle (0.35);

  % Label
  \node[below right, font=\small] at (0.2,5.2)
    {$t = \pgfmathprintnumber[fixed, precision=2]{\tval}$};
\end{tikzpicture}
\end{document}
