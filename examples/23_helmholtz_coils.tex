-- Elliptic integrals (Abramowitz & Stegun polynomial approximations)

function elliptic_K(k)
    local m = k * k
    if m >= 1.0 then m = 0.9999 end
    local m1 = 1.0 - m
    local a0, a1, a2, a3 = 1.3862944, 0.1119723, 0.0725296, 0.0140218
    local b0, b1, b2, b3 = 0.5, 0.1213478, 0.0288729, 0.0026070
    local logm1 = math.log(1.0 / m1)
    return (a0 + a1*m1 + a2*m1*m1 + a3*m1*m1*m1)
         + (b0 + b1*m1 + b2*m1*m1 + b3*m1*m1*m1) * logm1
end

function elliptic_E(k)
    local m = k * k
    if m >= 1.0 then m = 0.9999 end
    local m1 = 1.0 - m
    local a0, a1, a2, a3 = 1.0, 0.4630151, 0.1077812, 0.0174420
    local b0, b1, b2, b3 = 0.0, 0.2452727, 0.0412496, 0.0039444
    local logm1 = math.log(1.0 / m1)
    return (a0 + a1*m1 + a2*m1*m1 + a3*m1*m1*m1)
         + (b0 + b1*m1 + b2*m1*m1 + b3*m1*m1*m1) * logm1
end

function loop_field(z, rho, z0, R)
    local dz = z - z0
    local alpha2 = (R + rho)^2 + dz^2
    local beta2 = (R - rho)^2 + dz^2
    if alpha2 < 1e-12 then alpha2 = 1e-12 end
    if beta2 < 1e-12 then beta2 = 1e-12 end
    local alpha = math.sqrt(alpha2)
    local k2 = 1.0 - beta2 / alpha2
    local k = math.sqrt(math.max(k2, 0))
    local K = elliptic_K(k)
    local E = elliptic_E(k)
    local C = 1.0 / (2.0 * math.pi)
    local Bz = C / alpha * (K + (R*R - rho*rho - dz*dz) / beta2 * E)
    local Brho = 0
    if rho > 1e-6 then
        Brho = C * dz / (rho * alpha) * (-K + (R*R + rho*rho + dz*dz) / beta2 * E)
    end
    return Bz, Brho
end

function helmholtz_field(z, rho, d, R)
    local Bz1, Br1 = loop_field(z, rho, d, R)
    local Bz2, Br2 = loop_field(z, rho, -d, R)
    return Bz1 + Bz2, Br1 + Br2
end

-- Distance from a point to the nearest coil conductor
function dist_to_coils(z, rho, d, R)
    local d1 = math.sqrt((z - d)^2 + (rho - R)^2)
    local d2 = math.sqrt((z + d)^2 + (rho - R)^2)
    return math.min(d1, d2)
end

-- Trace a full field line (forward + backward) from a seed point.
function gen_full_field_line(seedZ, seedRho, d, R, steps, ds)
    local minCoilDist = 0.18
    local back = {}
    local z, rho = seedZ, seedRho
    for i = 1, steps do
        local Bz, Br = helmholtz_field(z, rho, d, R)
        local Bmag = math.sqrt(Bz*Bz + Br*Br)
        if Bmag < 1e-10 then break end
        local nz = z - Bz / Bmag * ds
        local nrho = rho - Br / Bmag * ds
        if math.abs(nz) > 4.2 or nrho > 3.0 or nrho < 0.04 then break end
        if dist_to_coils(nz, nrho, d, R) < minCoilDist then break end
        z, rho = nz, nrho
        back[#back + 1] = string.format("(%.5f, %.5f)", z, rho)
    end
    local parts = {}
    for i = #back, 1, -1 do
        parts[#parts + 1] = back[i]
    end
    parts[#parts + 1] = string.format("(%.5f, %.5f)", seedZ, seedRho)
    z, rho = seedZ, seedRho
    for i = 1, steps do
        local Bz, Br = helmholtz_field(z, rho, d, R)
        local Bmag = math.sqrt(Bz*Bz + Br*Br)
        if Bmag < 1e-10 then break end
        local nz = z + Bz / Bmag * ds
        local nrho = rho + Br / Bmag * ds
        if math.abs(nz) > 4.2 or nrho > 3.0 or nrho < 0.04 then break end
        if dist_to_coils(nz, nrho, d, R) < minCoilDist then break end
        z, rho = nz, nrho
        parts[#parts + 1] = string.format("(%.5f, %.5f)", z, rho)
    end
    if #parts >= 2 then
        tex.sprint(table.concat(parts, " "))
    else
        tex.sprint("(0,0) (0.001,0)")
    end
end

-- Generate auto-normalized Bz profile along the axis for the inset.
-- Outputs coordinates scaled so the peak fits within [0, targetH].
function gen_axis_profile_normalized(d, R, zmin, zmax, samples, targetH)
    local step = (zmax - zmin) / (samples - 1)
    local vals = {}
    local Bmax = 0
    for i = 0, samples - 1 do
        local z = zmin + i * step
        local Bz, _ = helmholtz_field(z, 0, d, R)
        vals[i] = Bz
        if Bz > Bmax then Bmax = Bz end
    end
    if Bmax < 1e-12 then Bmax = 1 end
    local scale = targetH / Bmax
    local parts = {}
    for i = 0, samples - 1 do
        local z = zmin + i * step
        parts[#parts + 1] = string.format("(%.5f, %.5f)", z, vals[i] * scale)
    end
    tex.sprint(table.concat(parts, " "))
end

-- Compute uniformity: max percentage deviation of Bz in central region
function compute_uniformity(d, R)
    local Bc, _ = helmholtz_field(0, 0, d, R)
    if math.abs(Bc) < 1e-12 then return 0 end
    local maxdev = 0
    for i = 0, 40 do
        local z = -0.4 * R + i * 0.8 * R / 40
        local Bz, _ = helmholtz_field(z, 0, d, R)
        local dev = math.abs((Bz - Bc) / Bc)
        if dev > maxdev then maxdev = dev end
    end
    return maxdev * 100
end

-- Heat map: emit TikZ fill rectangles colored by field strength
function gen_heatmap(d, R, xmin, xmax, ymin, ymax, nx, ny)
    local dx = (xmax - xmin) / nx
    local dy = (ymax - ymin) / ny
    local Bmax = 0
    local grid = {}
    for ix = 0, nx - 1 do
        grid[ix] = {}
        for iy = 0, ny - 1 do
            local z = xmin + (ix + 0.5) * dx
            local rho = math.abs(ymin + (iy + 0.5) * dy)
            local Bz, Br = helmholtz_field(z, rho, d, R)
            local Bmag = math.sqrt(Bz*Bz + Br*Br)
            grid[ix][iy] = Bmag
            if Bmag > Bmax then Bmax = Bmag end
        end
    end
    if Bmax < 1e-12 then Bmax = 1 end
    for ix = 0, nx - 1 do
        for iy = 0, ny - 1 do
            local z = xmin + ix * dx
            local rho = ymin + iy * dy
            local frac = math.min(grid[ix][iy] / Bmax, 1.0)
            local pct = math.floor(frac * 55 + 0.5)
            if pct > 2 then
                tex.sprint(string.format(
                    "\\fill[atlantic!%d!sandstorm, opacity=0.28] (%.4f, %.4f) rectangle (%.4f, %.4f);",
                    pct, z, rho, z + dx, rho + dy
                ))
            end
        end
    end
end
