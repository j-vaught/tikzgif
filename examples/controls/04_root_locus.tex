% Example 27: Root Locus Animation
% Animates poles of 1 + K*G(s) = 0 where G(s) = 1/(s(s+2)(s+4)).
% Three-pole system with breakaway, complex branches, and asymptotes.
% Parameter: \PARAM (0 to 1, mapped to loop gain K = 0 to 70)
% Render: tikzgif render 27_root_locus.tex --engine lualatex
% Difficulty: advanced
% Features demonstrated: root locus, cubic root solving, pole migration,
%   damping lines, breakaway point

\begin{luacode*}
function cubic_roots(K)
    local a, b, c = 6, 8, K
    local p = b - a*a/3
    local q = 2*a*a*a/27 - a*b/3 + c
    local disc = q*q/4 + p*p*p/27
    local shift = -a/3

    if disc > 1e-10 then
        local sqd = math.sqrt(disc)
        local u = -q/2 + sqd
        local v = -q/2 - sqd
        local cbrt_u = (u >= 0) and u^(1/3) or -((-u)^(1/3))
        local cbrt_v = (v >= 0) and v^(1/3) or -((-v)^(1/3))
        local t1 = cbrt_u + cbrt_v
        local re_complex = -t1/2 + shift
        local im_complex = math.abs(cbrt_u - cbrt_v) * math.sqrt(3)/2
        return t1 + shift, 0, re_complex, im_complex, re_complex, -im_complex
    elseif disc < -1e-10 then
        local r = math.sqrt(-p*p*p/27)
        local theta = math.acos(math.max(-1, math.min(1, -q/(2*r))))
        local cbrt_r = r^(1/3)
        local t1 = 2*cbrt_r * math.cos(theta/3)
        local t2 = 2*cbrt_r * math.cos((theta + 2*math.pi)/3)
        local t3 = 2*cbrt_r * math.cos((theta + 4*math.pi)/3)
        return t1 + shift, 0, t2 + shift, 0, t3 + shift, 0
    else
        local cbrt_q = (q >= 0) and (q/2)^(1/3) or -((-q/2)^(1/3))
        local t1 = -2*cbrt_q
        local t2 = cbrt_q
        return t1 + shift, 0, t2 + shift, 0, t2 + shift, 0
    end
end

function gen_all_branches(Kmax, steps)
    local branches = {{}, {}, {}}
    local prev = {{0, 0}, {-2, 0}, {-4, 0}}
    for i = 0, steps do
        local K = Kmax * i / steps
        local r1, i1, r2, i2, r3, i3 = cubic_roots(K)
        local roots = {{r1, i1}, {r2, i2}, {r3, i3}}
        local used = {false, false, false}
        local assignment = {0, 0, 0}
        for b = 1, 3 do
            local bestDist = 1e30
            local bestIdx = 1
            for j = 1, 3 do
                if not used[j] then
                    local dr = roots[j][1] - prev[b][1]
                    local di = roots[j][2] - prev[b][2]
                    local dist = dr*dr + di*di
                    if dist < bestDist then
                        bestDist = dist
                        bestIdx = j
                    end
                end
            end
            assignment[b] = bestIdx
            used[bestIdx] = true
        end
        for b = 1, 3 do
            local r = roots[assignment[b]]
            branches[b][#branches[b] + 1] = string.format("(%.5f, %.5f)", r[1], r[2])
            prev[b] = r
        end
    end
    return branches
end

function draw_locus_branches(Kmax, steps)
    local branches = gen_all_branches(Kmax, steps)
    for b = 1, 3 do
        if #branches[b] >= 2 then
            tex.sprint("\\draw[atlantic, thick] plot[smooth, tension=0.4] coordinates {"
                .. table.concat(branches[b], " ") .. "};")
        end
    end
end

function draw_poles(K)
    local r1, i1, r2, i2, r3, i3 = cubic_roots(K)
    local roots = {{r1, i1}, {r2, i2}, {r3, i3}}
    for _, r in ipairs(roots) do
        tex.sprint(string.format(
            "\\filldraw[garnet] (%.5f, %.5f) circle (3.5pt);",
            r[1], r[2]
        ))
    end
end
\end{luacode*}
