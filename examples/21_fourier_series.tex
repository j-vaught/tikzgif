% Example 21: Fourier Series Buildup
% Partial-sum convergence of a Fourier series to a square wave.
% Parameter: \PARAM (0 to 1, mapped to number of odd harmonics = 1 to 41)
% Render: tikzgif render 21_fourier_series.tex --engine lualatex
% Difficulty: intermediate
% Features demonstrated: Fourier series, convergence, Lua-accelerated math

\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usepackage{pgfplots}
\usepackage{amsmath}
\pgfplotsset{compat=1.18}
\usepgfplotslibrary{fillbetween}

% USC Brand Colors
\definecolor{garnet}{HTML}{73000A}
\definecolor{rose}{HTML}{CC2E40}
\definecolor{atlantic}{HTML}{466A9F}
\definecolor{congaree}{HTML}{1F414D}
\definecolor{horseshoe}{HTML}{65780B}
\definecolor{grass}{HTML}{CED318}
\definecolor{honeycomb}{HTML}{A49137}
\definecolor{warmgrey}{HTML}{676156}
\definecolor{sandstorm}{HTML}{FFF2E3}

\begin{document}
\begin{tikzpicture}

\pgfmathtruncatemacro{\Nterms}{\PARAM * 40 + 1}

%% ---- MAIN WAVEFORM PLOT ----
\begin{axis}[
    name=waveax,
    width=10cm, height=6cm,
    xmin=-3.5, xmax=10.5, ymin=-1.6, ymax=1.6,
    xlabel={$x$}, ylabel={$f(x)$},
    title={$N = \Nterms$},
    grid=major, grid style={warmgrey!20}, thick,
    every axis title/.style={font=\Large, at={(0.5,1.05)}},
    tick label style={font=\small},
    xtick={-3.14159, 0, 3.14159, 6.28318, 9.42478},
    xticklabels={$-\pi$, $0$, $\pi$, $2\pi$, $3\pi$},
    declare function={N=\Nterms;},
]
    % Target square wave
    \addplot[domain=-3.5:10.5, samples=500, name path=sqwave,
        dashed, black!40, thin]
        {(sin(deg(x)) >= 0) ? 1 : -1};

    % Fourier partial sum (lua backend accelerated via declare function)
    \addplot[domain=-3.5:10.5, samples=800, name path=fourier,
        atlantic, very thick]
        {(4/3.14159) * (
            (N >= 1  ? sin(deg(1*x))/1   : 0) +
            (N >= 3  ? sin(deg(3*x))/3   : 0) +
            (N >= 5  ? sin(deg(5*x))/5   : 0) +
            (N >= 7  ? sin(deg(7*x))/7   : 0) +
            (N >= 9  ? sin(deg(9*x))/9   : 0) +
            (N >= 11 ? sin(deg(11*x))/11 : 0) +
            (N >= 13 ? sin(deg(13*x))/13 : 0) +
            (N >= 15 ? sin(deg(15*x))/15 : 0) +
            (N >= 17 ? sin(deg(17*x))/17 : 0) +
            (N >= 19 ? sin(deg(19*x))/19 : 0) +
            (N >= 21 ? sin(deg(21*x))/21 : 0) +
            (N >= 23 ? sin(deg(23*x))/23 : 0) +
            (N >= 25 ? sin(deg(25*x))/25 : 0) +
            (N >= 27 ? sin(deg(27*x))/27 : 0) +
            (N >= 29 ? sin(deg(29*x))/29 : 0) +
            (N >= 31 ? sin(deg(31*x))/31 : 0) +
            (N >= 33 ? sin(deg(33*x))/33 : 0) +
            (N >= 35 ? sin(deg(35*x))/35 : 0) +
            (N >= 37 ? sin(deg(37*x))/37 : 0) +
            (N >= 39 ? sin(deg(39*x))/39 : 0) +
            (N >= 41 ? sin(deg(41*x))/41 : 0)
        )};

    % Error shading between partial sum and square wave
    \addplot[fill=rose, fill opacity=0.12] fill between[of=fourier and sqwave];

\end{axis}

\end{tikzpicture}
\end{document}
