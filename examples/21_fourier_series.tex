% Lua functions to generate coordinate lists
\begin{luacode*}
function gen_square_coords(xmin, xmax, samples)
    local step = (xmax - xmin) / (samples - 1)
    local parts = {}
    for i = 0, samples - 1 do
        local x = xmin + i * step
        local y = (math.sin(x) >= 0) and 1 or -1
        parts[#parts + 1] = string.format("(%.6f, %.6f)", x, y)
    end
    tex.sprint(table.concat(parts, " "))
end

function gen_fourier_coords(xmin, xmax, samples, N)
    local step = (xmax - xmin) / (samples - 1)
    local parts = {}
    for i = 0, samples - 1 do
        local x = xmin + i * step
        local s = 0
        for n = 1, N, 2 do
            s = s + math.sin(n * x) / n
        end
        local y = (4 / math.pi) * s
        parts[#parts + 1] = string.format("(%.6f, %.6f)", x, y)
    end
    tex.sprint(table.concat(parts, " "))
end
\end{luacode*}

    \addplot[name path=fourier, atlantic, very thick]
