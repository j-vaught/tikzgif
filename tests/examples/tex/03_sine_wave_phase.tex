% Example 03: Sine Wave Phase Shift
% A sine wave sliding horizontally across the screen.
% Parameter: \PARAM (phase shift in degrees, 0 to 360)
% Range: 0 to 360, recommended 60 frames (6-degree steps)
% Difficulty: beginner
% Features demonstrated: pgfplots integration, trigonometric functions
\documentclass[tikz]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    width=10cm, height=5cm,
    domain=0:720,
    samples=200,
    axis lines=middle,
    xlabel={$x$},
    ylabel={$y$},
    ymin=-1.5, ymax=1.5,
    xmin=0, xmax=720,
    xtick={0,90,180,270,360,450,540,630,720},
    xticklabels={$0$,$\frac{\pi}{2}$,$\pi$,$\frac{3\pi}{2}$,$2\pi$,
                 $\frac{5\pi}{2}$,$3\pi$,$\frac{7\pi}{2}$,$4\pi$},
    ytick={-1,0,1},
    grid=major,
    grid style={gray!30},
    title={\small Phase shift $\phi = \PARAM^\circ$},
    every axis title/.style={at={(0.5,1.05)}},
    tick label style={font=\tiny},
  ]
    \addplot[red!70!black, thick, smooth] {sin(x + \PARAM)};
    \addplot[blue!50!black, thick, smooth, dashed, opacity=0.3] {sin(x)};
  \end{axis}
\end{tikzpicture}
\end{document}
